<body>

    <h2>Student Majors</h2>

    <button onclick="loadData('CIT')">CIT Majors</button>
    <button onclick="loadData('BUS')">Business Majors</button>

    <div id="output"></div>

    <!-- Handlebars Template -->
    <script id="studentTemplate" type="text/x-handlebars-template">
        <table>
            <tr>
                <th>Name</th>
                <th>Major</th>
                <th>Midterm</th>
                <th>Final</th>
            </tr>

            {{#each students}}
            <tr>
                <td>{{name}}</td>
                <td>{{major}}</td>
                <td>{{midterm}}</td>
                <td>{{final}}</td>
            </tr>
            {{/each}}

        </table>
    </script>

    <script>
        function loadData(majorType) {

            fetch('cit5students.json')
                .then(response => response.json())
                .then(data => {

                    // Filter students
                    let filtered = data.filter(student => student.major === majorType);

                    // Handlebars
                    let source = document.getElementById("studentTemplate").innerHTML;
                    let template = Handlebars.compile(source);

                    let html = template({ students: filtered });

                    document.getElementById("output").innerHTML = html;
                })
                .catch(error => {
                    document.getElementById("output").innerHTML = "Error loading data.";
                    console.error(error);
                });
        }
    </script>
